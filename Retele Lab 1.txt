def XOR(Polinom, Message):
    for index in range(len(Polinom)):
        if Polinom[index] == Message[index]:
            Message = Message[:index] + '0' + Message[index + 1:]
        else:
            Message = Message[:index] + '1' + Message[index + 1:]
    return Message


def removeZeros(Message):
    index = 0
    while Message[index] == '0':
        index += 1
    Message = Message.replace('0', '', index)
    return Message
